# Review of C Programming Language
- READ C Programming Language Textbook
## Basic C Program

```c

#include <stdio.h>

  

int main() {

    printf("Hello, World!\n");

    return 0;

}

```

  

---

  

## Basic Data Types

- ****int****: integers

- ****float****: single-precision decimals

- ****double****: double-precision decimals

- ****char****: single characters

- ****void****: no value

  

```c

int age = 20;

float gpa = 3.5;

double pi = 3.14159;

char grade = 'A';

```

  

---

  

## Arrays

- Collection of elements of the same type.

- Declared with fixed size: `int arr[10];`

- Indexed from `0`.

  

```c

int arr[5] = {1, 2, 3, 4, 5};

printf("%d", arr[0]); // prints 1

```

  

---

  

## Functions

- Defined with return type, name, parameters.

- Can be declared before use with a prototype.

  

```c

int add(int a, int b) {

    return a + b;

}

  

int main() {

    int sum = add(3, 4);

    printf("%d", sum);

}

```

  

---

  

## Bitwise Operators

- `&` AND  

- `|` OR  

- `^` XOR  

- `~` NOT  

- `<<` Left shift  

- `>>` Right shift  

  

```c

int a = 5, b = 3;

printf("%d", a & b);

```

  

---

  

## Conditional Expressions

- Ternary operator: `condition ? expr1 : expr2`

  

```c

int x = 5;

int result = (x > 0) ? 1 : 0;

```

  

---

  

## Flow Control: if/else

```c

if (x > 0) {

    printf("Positive");

} else {

    printf("Non-positive");

}

```

  

---

  

## Flow Control: switch

```c

int choice = 2;

switch (choice) {

    case 1: printf("One"); break;

    case 2: printf("Two"); break;

    default: printf("Other");

}

```

  

---

  

## Loops

  

### for Loop

```c

for (int i = 0; i < 5; i++) {

    printf("%d ", i);

}

```

  

### while Loop

```c

int i = 0;

while (i < 5) {

    printf("%d ", i);

    i++;

}

```

  

### do-while Loop

```c

int i = 0;

do {

    printf("%d ", i);

    i++;

} while (i < 5);

```

  

---

  

## Loop Control

- ****break****: exit loop early.

- ****continue****: skip rest of current iteration.

  

```c

for (int i = 0; i < 5; i++) {

    if (i == 3) continue;

    if (i == 4) break;

    printf("%d ", i);

}

```

  

---

  

## Flow Control: goto and Labels

```c

int i = 0;

start:

if (i < 5) {

    printf("%d ", i);

    i++;

    goto start;

}

```

## Functions: Arguments & Return Values

```c

    return a * b;

    printf("%d", multiply(2, 3));
```

## Static Variables

```c

void counter() {

    static int count = 0;

    count++;

    printf("%d ", count);

}

  

int main() {

    counter();

    counter();

    counter();

}

```

  

---

  

## Register Variables

```c

void fast() {

    register int i;

    for (i = 0; i < 5; i++) {

        printf("%d ", i);

    }

}

```

  

---

  

## Scope of Variables

- ****Local****: inside a function/block.

- ****Global****: declared outside functions.

- ****Block scope****: within `{ }`.

  

```c

int globalVar = 10;

  

void func() {

    int localVar = 5;

    printf("%d", localVar);

}

```

  

---

  

## Recursion

```c

int factorial(int n) {

    if (n == 0) return 1;

    return n * factorial(n - 1);

}

```

  

---

  

## Pointers & Addresses

```c

int x = 10;

int *p = &x;

printf("%d", *p);

```

## Pointers and Function Arguments

```c

void update(int *a) {

    *a = *a + 5;

}

  

int main() {

    int x = 10;

    update(&x);

    printf("%d", x); // prints 15

}

```
